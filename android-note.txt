   sudo apt-get install gnome-panel

2. 安装 openjdk-7
   sudo apt-get install gnome-panel

3. 64位系统执行adb问题
   sudo apt-get install ia32-libs

4. android 5.0 KVM

   git config --global user.name "YuYongzhi:

   git config --global color.ui auto

   git push origin master
